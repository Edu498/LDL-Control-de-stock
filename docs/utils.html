 5class Eventos:
 6    """
 7    Módulo observador-sujeto que permite comunicar cambios en el sistema 
 8    sin acoplamiento directo entre módulos.
 9
10    Los controladores notifican eventos
11    y las interfaces (vistas) se suscriben para actualizarse automáticamente.
12    """
13    
14    _observadores = {}
15    
16    @classmethod
17    def suscribir(cls, evento, callback):
18        """
19        Suscribe un callback a un evento
20        Args:
21            evento (str): Nombre del evento al que se suscribe.
22            callback (function): Función a ejecutar cuando se notifique el evento.
23        """
24        if evento not in cls._observadores:
25            cls._observadores[evento] = []
26        cls._observadores[evento].append(callback)
27        
28    @classmethod
29    def desuscribir(cls, evento, callback):
30        """
31        Desuscribe un callback de un evento
32        Args:
33            evento (str): Nombre del evento.
34            callback (function): Función a remover.
35        """
36        if evento in cls._observadores and callback in cls._observadores[evento]:
37            cls._observadores[evento].remove(callback)
38    
39    @classmethod
40    def notificar(cls, evento, datos=None):
41        """
42        Notifica a todos los observadores de un evento
43        Args:
44            evento (str): Nombre del evento a notificar.
45            datos (any, optional): Datos adicionales a pasar a los callbacks.
46        """
47        if evento in cls._observadores:
48            observadores_activos = []
49            
50            for callback in cls._observadores[evento]:
51                # Limpieza automática si el callback pertenece a un Widget destruido de Tkinter
52                if hasattr(callback, '__self__'):
53                    instancia = callback.__self__
54                    if hasattr(instancia, 'window'):
55                        widget = instancia.window
56                        import tkinter as tk
57                        if isinstance(widget, tk.Widget):
58                            try:
59                                if not widget.winfo_exists():
60                                    continue  # Omitir y limpiar este observador
61                            except Exception:
62                                continue  # Omitir en caso de cualquier error de consulta
63                
64                observadores_activos.append(callback)
65                
66                try:
67                    if datos:
68                        callback(datos)
69                    else:
70                        callback()
71                except Exception as e:
72                    print(f"Error en callback: {e}")
73            
74            # Actualizar la lista con solo los observadores que siguen activos
75            cls._observadores[evento] = observadores_activos
76    
77    @classmethod
78    def limpiar(cls):
79        """
80        Limpia todos los observadores
81        """
82        cls._observadores = {}

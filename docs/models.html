 14    def __init__(self, id_producto=None, codigo="", nombre="", descripcion="",
 15                 id_categoria=None, id_proveedor=None, precio_compra=0.0,
 16                 precio_venta=0.0, stock_actual=0, stock_minimo=5, 
 17                 stock_maximo=None, ubicacion="", unidad_medida="unidad", activo=True):
 18        """
 19        Inicializa una nueva instancia de la clase Producto.
 20
 21        Args:
 22            id_producto (int, optional): Identificador único del producto en la base de datos.
 23            codigo (str, optional): Código de barras o identificador interno. Por defecto "".
 24            nombre (str, optional): Nombre del producto o insumo. Por defecto "".
 25            descripcion (str, optional): Detalles adicionales del artículo. Por defecto "".
 26            id_categoria (int, optional): ID de la categoría a la que pertenece (FK).
 27            id_proveedor (int, optional): ID del proveedor principal del artículo (FK).
 28            precio_compra (float, optional): Costo de adquisición del producto. Por defecto 0.0.
 29            precio_venta (float, optional): Precio de venta al público. Por defecto 0.0.
 30            stock_actual (int, optional): Cantidad física actual en el local/depósito. Por defecto 0.
 31            stock_minimo (int, optional): Umbral mínimo para disparar alertas de reposición. Por defecto 5.
 32            stock_maximo (int, optional): Límite de capacidad en depósito para este artículo.
 33            ubicacion (str, optional): Pasillo, estante o sector donde se almacena. Por defecto "".
 34            unidad_medida (str, optional): Ej: "unidad", "kg", "litros". Por defecto "unidad".
 35            activo (bool, optional): Indica si el producto sigue siendo comercializado/usado. Por defecto True.
 36        """
 37        self.id_producto = id_producto
 38        self.codigo = codigo
 39        self.nombre = nombre
 40        self.descripcion = descripcion
 41        self.id_categoria = id_categoria
 42        self.id_proveedor = id_proveedor
 43        self.precio_compra = float(precio_compra)
 44        self.precio_venta = float(precio_venta)
 45        self.stock_actual = int(stock_actual)
 46        self.stock_minimo = int(stock_minimo)
 47        self.stock_maximo = int(stock_maximo) if stock_maximo else None
 48        self.ubicacion = ubicacion
 49        self.unidad_medida = unidad_medida
 50        self.activo = activo
 51    
 52    @property
 53    def estado_stock(self):
 54        """
 55        Evalúa el stock actual y determina la situación del articulo.
 56
 57        Returns:
 58            str: Devuelve el estado del stock ('SIN STOCK', 'STOCK BAJO' o 'NORMAL')
 59        """
 60        if self.stock_actual <= 0:
 61            return ("SIN STOCK")
 62        elif self.stock_actual <= self.stock_minimo:
 63            return ("STOCK BAJO")
 64        else:
 65            return ("NORMAL")
 66    
 67    @property
 68    def necesita_reposicion(self):
 69        """
 70        Verifica si el articulo alcanzo o perforo el stock minimo.
 71
 72        Returns:
 73            bool: True si necesita reposicion, False en caso contrario.
 74        """
 75        return self.stock_actual <= self.stock_minimo
 76    
 77    @property
 78    def cantidad_recomendada(self):
 79        """
 80        Calcula la cantidad de unidades que deberian ordenarse para alcanzar el stock maximo.
 81        
 82        Returns:
 83            int: La diferencia entre el stock maximo y el stock actual. Devuelve 0 si el stock es normal.
 84         """
 85        if self.stock_actual < self.stock_minimo:
 86            return self.stock_minimo - self.stock_actual
 87        return 0

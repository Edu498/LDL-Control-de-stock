115        """
116        Actualiza el stock de un producto y registra el movimiento
117        
118        Args:
119            id_producto: ID del producto
120            cantidad: Cantidad a modificar (POSITIVA para ENTRADA, NEGATIVA para SALIDA)
121            tipo_movimiento: 1=Venta, 2=Compra, 3=Ajuste+, 4=Ajuste-
122            usuario: Usuario que realiza la operación
123            referencia: Información adicional de referencia
124        """
125        # ========== DEBUG ==========
126        print(f"\n📊 STOCK CONTROLLER:")
127        print(f"   Producto ID: {id_producto}")
128        print(f"   Cantidad: {cantidad} ({'ENTRADA' if cantidad > 0 else 'SALIDA'})")
129        print(f"   Tipo Movimiento: {tipo_movimiento}")
130        print(f"   Usuario: {usuario}")
131        # ===========================
132        
133        conexion = None
134        cursor = None
135        
136        try:
137            conexion = get_connection()
138            cursor = conexion.cursor()
139            conexion.start_transaction()
140            
141            # Obtener stock actual con bloqueo para actualización
142            cursor.execute("SELECT stock_actual, nombre FROM productos WHERE id_producto = %s FOR UPDATE", (id_producto,))
143            resultado = cursor.fetchone()
144            
145            if not resultado:
146                raise ValueError(f"Producto con ID {id_producto} no encontrado")
147            
148            stock_actual = resultado[0]
149            nombre_producto = resultado[1]
150            stock_nuevo = stock_actual + cantidad
151            
152            # Validación de stock negativo
153            if stock_nuevo < 0:
154                raise ValueError(f"Stock insuficiente. Stock actual: {stock_actual}, intenta quitar: {-cantidad}")
155            
156            # Actualizar stock del producto
157            cursor.execute("""
158                UPDATE productos 
159                SET stock_actual = %s 
160                WHERE id_producto = %s
161            """, (stock_nuevo, id_producto))
162            
163            # Registrar movimiento en historial
164            motivo = referencia.get('motivo', 'Ajuste manual') if referencia else 'Ajuste manual'
165            referencia_tipo = referencia.get('tipo') if referencia else 'ajuste'
166            referencia_id = referencia.get('id') if referencia else None
167            
168            cursor.execute("""
169                INSERT INTO movimientos_stock (id_producto, id_tipo_movimiento, cantidad, 
170                                               stock_antes, stock_despues, referencia_tipo, 
171                                               referencia_id, usuario, fecha)
172                VALUES (%s, %s, %s, %s, %s, %s, %s, %s, NOW())
173            """, (id_producto, tipo_movimiento, cantidad, stock_actual, stock_nuevo,
174                  referencia_tipo, referencia_id, usuario))
175            
176            conexion.commit()
177            print(f"✅ Stock actualizado: {nombre_producto} - {stock_actual} → {stock_nuevo} (cantidad: {cantidad})")
178            if cantidad > 0:
179                print(f"   → TIPO: ENTRADA")
180            elif cantidad < 0:
181                print(f"   → TIPO: SALIDA")
182            return stock_nuevo
183

186                conexion.rollback()
187            print(f"❌ Error al actualizar stock: {e}")

 13    def registrar_venta(venta):
 14        """
 15        Registra una nueva venta de forma transaccional (ACID). Inserta la cabecera,
 16        los detalles, descuenta el inventario físico y genera el historial de movimientos.
 17        
 18        Args:
 19            venta (Venta): Objeto que contiene los datos de la venta y sus detalles.
 20            
 21        Returns:
 22            Venta: El objeto venta actualizado con su ID generado por la base de datos.
 23            
 24        Raises:
 25            ValueError: Si se intenta vender más stock del disponible.
 26        """
 27        conexion = None
 28        cursor = None
 29        
 30        try:
 31            conexion = get_connection()
 32            cursor = conexion.cursor()
 33            conexion.start_transaction()
 34            
 35            subtotal = venta.subtotal
 36            iva = venta.iva
 37            total = venta.total
 38            
 39            cursor.execute("""
 40                INSERT INTO ventas (numero_factura, cliente_nombre, subtotal, iva, total, usuario, id_estado, fecha_venta)
 41                VALUES (%s, %s, %s, %s, %s, %s, 2, NOW())
 42            """, (venta.numero_factura, venta.cliente_nombre, subtotal, iva, total, venta.usuario))
 43            
 44            venta.id_venta = cursor.lastrowid
 45            
 46            for detalle in venta.detalles:
 47                cursor.execute("""
 48                    INSERT INTO detalles_venta (id_venta, id_producto, cantidad, 
 49                                                precio_unitario, subtotal)
 50                    VALUES (%s, %s, %s, %s, %s)
 51                """, (venta.id_venta, detalle.id_producto, detalle.cantidad,
 52                      detalle.precio_unitario, detalle.subtotal))
 53                
 54                cursor.execute("SELECT stock_actual FROM productos WHERE id_producto = %s FOR UPDATE", (detalle.id_producto,))
 55                resultado_stock = cursor.fetchone()
 56                
 57                if not resultado_stock:
 58                    raise ValueError(f"Producto con ID {detalle.id_producto} no encontrado.")
 59                    
 60                stock_actual = resultado_stock[0]
 61                stock_nuevo = stock_actual - detalle.cantidad
 62                
 63                if stock_nuevo < 0:
 64                    raise ValueError(f"Stock insuficiente para descontar {detalle.cantidad} unidades.")
 65
 66                cursor.execute("""
 67                    UPDATE productos 
 68                    SET stock_actual = %s 
 69                    WHERE id_producto = %s
 70                """, (stock_nuevo, detalle.id_producto))
 71                
 72                cursor.execute("""
 73                    INSERT INTO movimientos_stock (id_producto, id_tipo_movimiento, cantidad,
 74                                                   stock_antes, stock_despues, referencia_tipo,
 75                                                   referencia_id, usuario, fecha)
 76                    VALUES (%s, 1, %s, %s, %s, 'venta', %s, %s, NOW())
 77                """, (detalle.id_producto, -detalle.cantidad, stock_actual, stock_nuevo, venta.id_venta, venta.usuario))
 78            
 79            conexion.commit()
 80            print(f"✅ Venta {venta.numero_factura} registrada - Total: ${total:.2f}")
 81            return venta
